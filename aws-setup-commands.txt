# AWS Setup Commands for Diamond District
# Run these commands on your LOCAL machine where AWS is configured

# Step 1: Create the s3-cors-policy.json file locally
cat > s3-cors-policy.json << 'EOF'
{
  "CORSRules": [
    {
      "AllowedHeaders": ["*"],
      "AllowedMethods": ["GET", "PUT", "POST", "DELETE", "HEAD"],
      "AllowedOrigins": ["https://watch.zerotodiamond.com", "http://localhost:3000"],
      "ExposeHeaders": ["ETag"],
      "MaxAgeSeconds": 3000
    }
  ]
}
EOF

# Step 2: Create the iam-policy.json file locally
cat > iam-policy.json << 'EOF'
{
  "Version": "2012-10-17",
  "Statement": [
    {
      "Effect": "Allow",
      "Action": [
        "s3:PutObject",
        "s3:GetObject",
        "s3:DeleteObject",
        "s3:PutObjectAcl",
        "s3:GetObjectAcl"
      ],
      "Resource": "arn:aws:s3:::diamond-district-videos/*"
    },
    {
      "Effect": "Allow",
      "Action": ["s3:ListBucket"],
      "Resource": "arn:aws:s3:::diamond-district-videos"
    }
  ]
}
EOF

# Step 3: Apply CORS configuration to your S3 bucket
aws s3api put-bucket-cors --bucket diamond-district-videos --cors-configuration file://s3-cors-policy.json

# Step 4: Create IAM user for S3 access
aws iam create-user --user-name diamond-district-s3

# Step 5: Attach policy to the user
aws iam put-user-policy --user-name diamond-district-s3 --policy-name DiamondDistrictS3Access --policy-document file://iam-policy.json

# Step 6: Create access key (SAVE THE OUTPUT!)
aws iam create-access-key --user-name diamond-district-s3

# IMPORTANT: Save the AccessKeyId and SecretAccessKey from the output above!
# You'll need them for the .env.local file on the server
